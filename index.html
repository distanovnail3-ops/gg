    function shuffle(arr) {
      const copy = [...arr];
      for (let i = copy.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [copy[i], copy[j]] = [copy[j], copy[i]];
      }
      return copy;
    }

    function getSelectedCategories() {
      const checks = document.querySelectorAll(".category-check");
      return [...checks].filter(c => c.checked).map(c => c.value);
    }

    function buildCategories() {
      Object.keys(DATA).forEach(category => {
        const id = "cat_" + category.replace(/\s+/g, "_");
        const label = document.createElement("label");
        label.className = "check";
        label.innerHTML = `
          <input type="checkbox" class="category-check" id="${id}" value="${category}">
          <span>${category} (${DATA[category].length})</span>
        `;
        categoriesContainer.appendChild(label);
      });
    }

    function generateWrongVariant(correctWord, plainWord) {
      const pool = [];
      Object.values(DATA).flat().forEach(item => {
        if (item.plain.toLowerCase() === plainWord.toLowerCase() && item.word !== correctWord) {
          pool.push(item.word);
        }
      });

      if (pool.length > 0) {
        return pool[Math.floor(Math.random() * pool.length)];
      }

      const vowels = "аеёиоуыэюяАЕЁИОУЫЭЮЯ";
      const positions = [];
      for (let i = 0; i < plainWord.length; i++) {
        if (vowels.includes(plainWord[i])) positions.push(i);
      }

      if (positions.length <= 1) {
        return correctWord;
      }

      const stressedIndex = [...correctWord].findIndex(ch => ch === ch.toUpperCase() && vowels.includes(ch));
      const lower = plainWord.toLowerCase();
      const variants = [];

      for (const pos of positions) {
        const chars = lower.split("");
        chars[pos] = chars[pos].toUpperCase();
        const candidate = chars.join("");
        if (candidate !== correctWord) variants.push(candidate);
      }

      if (variants.length) {
        return variants[Math.floor(Math.random() * variants.length)];
      }
      return correctWord;
    }

    function makeOptions(question) {
      const options = new Set();
      options.add(question.word);

      while (options.size < 4) {
        let variant = generateWrongVariant(question.word, question.plain);
        if (variant === question.word) {
          const allWords = Object.values(DATA).flat().map(x => x.word);
          variant = allWords[Math.floor(Math.random() * allWords.length)];
        }
        options.add(variant);
      }

      return shuffle([...options]);
    }

    function updateStats() {
      correctCountEl.textContent = correctCount;
      wrongCountEl.textContent = wrongCount;
      remainingCountEl.textContent = Math.max(0, currentPool.length - currentIndex);
      modeLabel.textContent = currentMode === "mistakes" ? "Повтор ошибок" : "Обычный";
      progressEl.textContent = currentPool.length
        ? `Слово ${Math.min(currentIndex + 1, currentPool.length)} из ${currentPool.length}`
        : "";
    }

    function showQuestion() {
      if (currentIndex >= currentPool.length) {
        questionWord.textContent = "Тренировка завершена";
        questionMeta.textContent = "";
        answersEl.innerHTML = "";
        resultEl.textContent = `Итог: правильно ${correctCount}, ошибок ${wrongCount}.`;
        nextBtn.disabled = true;
        updateStats();
        return;
      }

      answered = false;
      nextBtn.disabled = true;
      resultEl.textContent = "";

      currentQuestion = currentPool[currentIndex];
      questionWord.textContent = currentQuestion.plain;
      questionMeta.textContent = currentQuestion.note || "Выбери правильное ударение";

      const options = makeOptions(currentQuestion);
      answersEl.innerHTML = "";

      options.forEach(option => {
        const btn = document.createElement("button");
        btn.className = "answer-btn";
        btn.textContent = option;
        btn.type = "button";
        btn.onclick = () => checkAnswer(btn, option);
        answersEl.appendChild(btn);
      });

      updateStats();
    }

    function checkAnswer(button, selected) {
      if (answered) return;
      answered = true;

      const buttons = document.querySelectorAll(".answer-btn");
      buttons.forEach(btn => btn.disabled = true);

      const correct = currentQuestion.word;

      if (selected === correct) {
        button.classList.add("correct");
        resultEl.textContent = "Верно!";
        correctCount++;
      } else {
        button.classList.add("wrong");
        wrongCount++;
        resultEl.textContent = `Неверно. Правильно: ${correct}`;
        if (!mistakes.some(m => m.word === currentQuestion.word && m.plain === currentQuestion.plain)) {
          mistakes.push(currentQuestion);
        }

        buttons.forEach(btn => {
          if (btn.textContent === correct) {
            btn.classList.add("correct");
          }
        });
      }

      nextBtn.disabled = false;
      updateStats();
    }

    function startQuiz(mode = "normal") {
      const selectedCategories = getSelectedCategories();

      if (mode === "normal" && selectedCategories.length === 0) {
        alert("Выбери хотя бы один раздел.");
        return;
      }

      if (mode === "mistakes") {
        if (mistakes.length === 0) {
          alert("Ошибок пока нет.");
          return;
        }
        currentPool = shuffle([...mistakes]);
      } else {
        let pool = [];
        selectedCategories.forEach(cat => {
          pool = pool.concat(DATA[cat]);
        });
        currentPool = shuffle(pool);
      }

      currentMode = mode;
      currentIndex = 0;
      correctCount = 0;
      wrongCount = 0;

      setupCard.classList.add("hidden");
      quizCard.classList.remove("hidden");

      showQuestion();
    }

    nextBtn.addEventListener("click", () => {
      currentIndex++;
      showQuestion();
    });

    backBtn.addEventListener("click", () => {
      quizCard.classList.add("hidden");
      setupCard.classList.remove("hidden");
    });

    restartBtn.addEventListener("click", () => {
      startQuiz(currentMode);
    });

    startBtn.addEventListener("click", () => startQuiz("normal"));
    mistakesBtn.addEventListener("click", () => startQuiz("mistakes"));
